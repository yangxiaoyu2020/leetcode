class Solution {
    // 数组中的最小最大值 计算
    public int minEatingSpeed(int[] piles, int h) {
        int left = 1;
        int right = 1_000_000_000;
        while (left < right) {
            int mid = (left + right) >>> 1;
            if (canDo(mid, piles, h)) {
                right = mid;
            }
            else {
                left = mid + 1;
            }
        }
        return left;
    }

    private boolean canDo(int speed, int[] piles, int h) {
        int need = 0;
        for (int p: piles) {
            need += (p-1) / speed + 1;
        }
        return need <= h;

    }
}
